	각 이야기꾼은 공포영화에 나올만한 일반적인 등장인물로 설정합니다. 예를 들어, 아래와 같은 역할을 받을 수 있습니다:
	
	\section{주정뱅이}
	
	\begin{lite}{알코올 중독}
		\entry{술... 술을 내놔!}
	\end{lite}
	
	\begin{lite}{술병}
		\entry{보드카로 가득찬 금속 플라스크}
	\end{lite}
	
	\section{꼬마 아이}
	
	\begin{lite}{극도의 무해함}
		\entry{유령 무서워...}
		\entry{작고 연약한 신체}
	\end{lite}
	
	\bigskip
	
	자신이 원하는 역할으로 설정하세요!
